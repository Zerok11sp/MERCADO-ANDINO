                                    </button>
                                </div>
                                <p className="text-xs text-stone-500 mb-3 leading-relaxed">¿Tienes dudas o inconvenientes? Escríbenos directamente por WhatsApp:</p>
                                <div className="space-y-2">
                                    <button 
                                        onClick={() => openSupport('51903095997')}
                                        className="w-full bg-emerald-50 hover:bg-emerald-100 text-emerald-800 text-xs font-semibold py-2.5 px-3 rounded-xl flex items-center justify-between transition-colors border border-emerald-200/60"
                                    >
                                        <span>Asistencia Soporte 1</span>
                                        <i data-lucide="chevron-right" className="w-3.5 h-3.5 text-emerald-600"></i>
                                    </button>
                                    <button 
                                        onClick={() => openSupport('51938612470')}
                                        className="w-full bg-emerald-50 hover:bg-emerald-100 text-emerald-800 text-xs font-semibold py-2.5 px-3 rounded-xl flex items-center justify-between transition-colors border border-emerald-200/60"
                                    >
                                        <span>Asistencia Soporte 2</span>
                                        <i data-lucide="chevron-right" className="w-3.5 h-3.5 text-emerald-600"></i>
                                    </button>
                                </div>
                            </div>
                        )}
                        <button
                            onClick={() => setIsHelpOpen(!isHelpOpen)}
                            className="bg-amber-600 hover:bg-amber-700 text-white w-14 h-14 rounded-full flex items-center justify-center shadow-2xl transition-all transform hover:scale-110 active:scale-95 border-2 border-white"
                            title="Ayuda y Soporte"
                        >
                            <i data-lucide="headphones" className="w-6 h-6"></i>
                        </button>
                    </div>

                    {/* MODAL: DETALLE DEL PRODUCTO */}
                    {selectedProduct && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 fade-in">
                            <div onClick={() => setSelectedProduct(null)} className="absolute inset-0 bg-black/60 backdrop-blur-sm"></div>
                            <div className="bg-white rounded-3xl w-full max-w-2xl overflow-hidden shadow-2xl relative z-10 zoom-in flex flex-col md:flex-row max-h-[90vh]">
                                <div className="w-full md:w-1/2 h-52 md:h-auto bg-stone-200 relative">
                                    <img src={selectedProduct.image} alt={selectedProduct.name} className="w-full h-full object-cover" />
                                    {selectedProduct.isExample && (
                                        <span className="absolute bottom-3 left-3 bg-orange-600 text-white text-[10px] font-black px-2 py-1 rounded shadow-md tracking-wider uppercase">
                                            EJEMPLO DE VENTA
                                        </span>
                                    )}
                                </div>
                                <div className="w-full md:w-1/2 p-6 flex flex-col justify-between overflow-y-auto">
                                    <div>
                                        <div className="flex justify-between items-start gap-2 mb-2">
                                            <span className="bg-amber-100 text-amber-900 text-xs font-bold px-2.5 py-1 rounded-full uppercase">{selectedProduct.category}</span>
                                            <button onClick={() => setSelectedProduct(null)} className="text-stone-400 hover:text-stone-600 p-1 rounded-full hover:bg-stone-100 transition-colors">
                                                <i data-lucide="x" className="w-5 h-5"></i>
                                            </button>
                                        </div>
                                        <h2 className="text-xl font-bold text-stone-900 mb-2 leading-snug">{selectedProduct.name}</h2>
                                        <div className="text-2xl font-black text-emerald-700 mb-4">S/. {selectedProduct.price.toFixed(2)}</div>
                                        <div className="space-y-3 border-t border-stone-100 pt-3 text-sm text-stone-600">
                                            <p><strong className="text-stone-800 block mb-0.5">Descripción:</strong> {selectedProduct.description || 'Sin descripción adicional.'}</p>
                                            {selectedProduct.origin && <p><strong>Origen:</strong> {selectedProduct.origin}</p>}
                                            <p className="flex items-center gap-1.5 font-medium text-stone-700"><i data-lucide="user" className="w-4 h-4 text-amber-600"></i> Productor: {selectedProduct.seller}</p>
                                        </div>
                                    </div>
                                    <div className="mt-6">
                                        <button 
                                            disabled={selectedProduct.isExample}
                                            onClick={(e) => contactSeller(selectedProduct, e)} 
                                            className={`w-full py-3 rounded-xl font-bold flex items-center justify-center gap-2 text-sm shadow-md transition-transform transform active:scale-95 ${
                                                selectedProduct.isExample 
                                                    ? 'bg-stone-200 text-stone-400 cursor-not-allowed' 
                                                    : 'bg-emerald-600 hover:bg-emerald-700 text-white'
                                            }`}
                                        >
                                            <i data-lucide="phone" className="w-4 h-4"></i> {selectedProduct.isExample ? 'Contacto Desactivado (Demo)' : 'Contactar por WhatsApp'}
                                        </button>
                                    </div>
                                </div>
                            </div>
                        </div>
                    )}

                    {/* MODAL: REGISTRO */}
                    {isAuthModalOpen && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 fade-in">
                            <div onClick={() => setIsAuthModalOpen(false)} className="absolute inset-0 bg-black/50 backdrop-blur-sm"></div>
                            <div className="bg-white rounded-2xl w-full max-w-sm p-6 shadow-xl relative z-10 zoom-in">
                                <div className="flex justify-between items-center mb-4">
                                    <h2 className="text-xl font-bold text-stone-900">Crear Cuenta</h2>
                                    <button onClick={() => setIsAuthModalOpen(false)} className="text-stone-400 hover:text-stone-600"><i data-lucide="x" className="w-5 h-5"></i></button>
                                </div>
                                <form onSubmit={handleRegister} className="space-y-4">
                                    <input type="text" required placeholder="Nombre Completo" className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={authForm.name} onChange={e => setAuthForm({...authForm, name: e.target.value})} />
                                    <input type="tel" required placeholder="WhatsApp (Ej. +51999888777)" className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={authForm.phone} onChange={e => setAuthForm({...authForm, phone: e.target.value})} />
                                    <input type="email" required placeholder="Correo Electrónico" className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={authForm.email} onChange={e => setAuthForm({...authForm, email: e.target.value})} />
                                    <button type="submit" className="w-full bg-amber-600 hover:bg-amber-700 text-white font-bold py-2.5 rounded-xl shadow-md transition-all">Registrarme</button>
                                </form>
                            </div>
                        </div>
                    )}

                    {/* MODAL: VENDER */}
                    {isSellModalOpen && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 fade-in">
                            <div onClick={() => setIsSellModalOpen(false)} className="absolute inset-0 bg-black/50 backdrop-blur-sm"></div>
                            <div className="bg-white rounded-2xl w-full max-w-md p-6 shadow-xl relative z-10 overflow-y-auto max-h-[90vh] zoom-in">
                                <div className="flex justify-between items-center mb-4">
                                    <h2 className="text-xl font-bold text-stone-900">Publicar Producto</h2>
                                    <button onClick={() => setIsSellModalOpen(false)} className="text-stone-400 hover:text-stone-600"><i data-lucide="x" className="w-6 h-6"></i></button>
                                </div>
                                <form onSubmit={handleCreateProduct} className="space-y-3.5">
                                    <input type="text" required placeholder="Nombre del Producto" className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={newProduct.name} onChange={e => setNewProduct({...newProduct, name: e.target.value})} />
                                    <div className="grid grid-cols-2 gap-3">
                                        <select className="w-full p-2.5 border rounded-xl text-sm bg-white focus:outline-amber-600" value={newProduct.category} onChange={e => setNewProduct({...newProduct, category: e.target.value})}>
                                            {CATEGORIES.filter(c => c !== 'Todos').map(c => <option key={c} value={c}>{c}</option>)}
                                        </select>
                                        <input type="number" step="0.01" required placeholder="Precio (S/.)" className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={newProduct.price} onChange={e => setNewProduct({...newProduct, price: e.target.value})} />
                                    </div>
                                    <input type="text" placeholder="Lugar de Procedencia" className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={newProduct.origin} onChange={e => setNewProduct({...newProduct, origin: e.target.value})} />
                                    
                                    {!currentUser && (
                                        <div className="p-3 bg-amber-50 rounded-xl border border-amber-200 space-y-2">
                                            <input type="text" required placeholder="Tu Nombre" className="w-full p-2 border rounded-lg text-sm bg-white" value={newProduct.seller} onChange={e => setNewProduct({...newProduct, seller: e.target.value})} />
                                            <input type="tel" required placeholder="Número de WhatsApp" className="w-full p-2 border rounded-lg text-sm bg-white" value={newProduct.phone} onChange={e => setNewProduct({...newProduct, phone: e.target.value})} />
                                        </div>
                                    )}

                                    <textarea rows="2" placeholder="Descripción detallada del producto..." className="w-full p-2.5 border rounded-xl text-sm focus:outline-amber-600" value={newProduct.description} onChange={e => setNewProduct({...newProduct, description: e.target.value})}></textarea>
                                    
                                    <div className="flex items-center gap-3">
                                        <label className="flex-1 flex flex-col items-center justify-center p-4 border-2 border-dashed border-stone-300 rounded-xl cursor-pointer hover:bg-stone-50 transition-colors">
                                            <i data-lucide="upload-cloud" className="w-6 h-6 text-stone-400 mb-1"></i>
                                            <span className="text-xs text-stone-500 font-medium">Subir Foto</span>
                                            <input type="file" accept="image/*" onChange={handleImageChange} className="hidden" />
                                        </label>
                                        {newProduct.image && <img src={newProduct.image} className="w-16 h-16 rounded-xl object-cover border shadow-sm flex-shrink-0" alt="preview" />}
                                    </div>

                                    <button type="submit" className="w-full bg-amber-600 hover:bg-amber-700 text-white font-bold py-3 rounded-xl shadow-md text-sm transition-all transform active:scale-95">Subir al Mercado</button>
                                </form>
                            </div>
                        </div>
                    )}

                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
